💻 OpenCode IDE

// Wispe™ Agent Framework
// Nanobot Auto-Classifier

const CATEGORIES = {
  notiz:   { icon: '📋', label: 'Notiz',   keywords: ['todo','aufgabe','merken'] },
  idee:    { icon: '💡', label: 'Idee',    keywords: ['idee','was wäre','könnte'] },
  meeting: { icon: '🎙️', label: 'Meeting', keywords: ['treffen','call','besprochen'] },
  archiv:  { icon: '📦', label: 'Archiv',  keywords: [] }
};

function classify(transcript) {
  for (const [cat, data] of Object.entries(CATEGORIES)) {
    if (data.keywords.some(k => transcript.toLowerCase().includes(k))) {
      return cat;
    }
  }
  return 'notiz'; // default
}

CREATE TABLE wispe_notes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  category TEXT CHECK (category IN
    ('notiz','idee','meeting','archiv')),
  content TEXT NOT NULL,
  speaker TEXT,
  device TEXT DEFAULT 'desktop',
  created_at TIMESTAMPTZ DEFAULT now(),
  opened_at TIMESTAMPTZ DEFAULT now(),
  archived_at TIMESTAMPTZ,
  cloud_archived BOOLEAN DEFAULT false
);

ALTER TABLE wispe_notes ENABLE ROW LEVEL SECURITY;
CREATE INDEX idx_opened ON wispe_notes(opened_at);
CREATE INDEX idx_category ON wispe_notes(category);
