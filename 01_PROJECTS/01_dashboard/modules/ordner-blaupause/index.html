🤖 Script 1: Auto-Sortierer (Inbox → Ordner)

Trigger: Alle 15 Min — sortiert neue Dateien aus 00_INBOX

// ⚠️ SICHERHEITS-PRINZIP: IMMER Kopie → Verifizieren → Dann erst entfernen!
// NIE removeFile() ohne vorherige Prüfung ob Kopie existiert!

function autoSortInbox() {
  const RULES = [
    { match: /image\//,        folder: '03_MEDIA/Screenshots' },
    { match: /video\//,        folder: '03_MEDIA/Videos' },
    { match: /audio\//,        folder: '03_MEDIA/Musik' },
    { match: /pdf|document/,   folder: '02_RESEARCH/docs' },
    { match: /zip|rar|7z/,     folder: '04_SYSTEM/00_Backups' },
    { match: /javascript|json|html|css/, folder: '01_PROJECTS/_code_inbox' }
  ];

  const inbox = findFolder('00_INBOX');
  if (!inbox) return;
  const files = inbox.getFiles();

  while (files.hasNext()) {
    const file = files.next();
    const mime = file.getMimeType();
    const name = file.getName();

    for (const rule of RULES) {
      if (rule.match.test(mime) || rule.match.test(name)) {
        const target = findOrCreate(rule.folder);

        // SCHRITT 1: Kopie in Zielordner erstellen
        target.addFile(file);

        // SCHRITT 2: VERIFIZIEREN dass Datei im Ziel angekommen ist
        const verify = target.getFilesByName(name);
        if (verify.hasNext()) {
          // SCHRITT 3: Erst JETZT aus Inbox entfernen
          inbox.removeFile(file);
          log('✅ Sortiert', name, rule.folder);
        } else {
          // FEHLER: Datei NICHT entfernen wenn Kopie fehlt!
          log('❌ FEHLER — Datei blieb in Inbox', name, rule.folder);
        }
        break;
      }
    }
  }
}

function findFolder(path) {
  const parts = path.split('/');
  let current = DriveApp.getRootFolder();
  for (const p of parts) {
    const iter = current.getFoldersByName(p);
    if (!iter.hasNext()) return null;
    current = iter.next();
  }
  return current;
}

function findOrCreate(path) {
  const parts = path.split('/');
  let current = DriveApp.getRootFolder();
  for (const p of parts) {
    const iter = current.getFoldersByName(p);
    current = iter.hasNext() ? iter.next() : current.createFolder(p);
  }
  return current;
}

function log(action, file, dest) {
  const sheet = SpreadsheetApp.openById('DEINE_SHEET_ID')
    .getSheetByName('Log') || SpreadsheetApp.openById('DEINE_SHEET_ID')
    .insertSheet('Log');
  sheet.appendRow([new Date(), action, file, dest]);
}

🤖 Script 2: Wöchentlicher Report + Alte Dateien → Archiv

function weeklyCleanup() {
  const ARCHIVE_AFTER_DAYS = 90;
  const now = new Date();
  // Exakt gleiche Ordner wie C:\DEVKiTZ\
  const folders = ['00_INBOX','01_PROJECTS','02_RESEARCH',
    '03_MEDIA','04_SYSTEM','05_INTERN','[DEEPKEEP]',
    '[WORKSPACE]','99_ARCHIVE'];

  let report = '📊 DkZ Drive-Report\n\n';
  let archived = 0;

  folders.forEach(name => {
    const folder = findFolder(name);
    if (!folder) return;
    const count = countFiles(folder);
    report += `${name}: ${count} Dateien\n`;
  });

  // Alte Inbox-Dateien → Archiv (MIT VERIFY!)
  const inbox = findFolder('00_INBOX');
  if (inbox) {
    const files = inbox.getFiles();
    while (files.hasNext()) {
      const f = files.next();
      const age = (now - f.getLastUpdated()) / 86400000;
      if (age > ARCHIVE_AFTER_DAYS) {
        const arch = findOrCreate('99_ARCHIVE/auto-archived');
        arch.addFile(f);
        // VERIFY vor Remove!
        const verify = arch.getFilesByName(f.getName());
        if (verify.hasNext()) {
          inbox.removeFile(f);
          archived++;
        }
      }
    }
  }

  report += `\nArchiviert: ${archived} alte Dateien`;
  MailApp.sendEmail({
    to: Session.getActiveUser().getEmail(),
    subject: '📊 DkZ Drive Report — ' + now.toLocaleDateString('de'),
    body: report
  });
}

function countFiles(folder) {
  let c = 0;
  const f = folder.getFiles();
  while (f.hasNext()) { f.next(); c++; }
  const s = folder.getFolders();
  while (s.hasNext()) { c += countFiles(s.next()); }
  return c;
}

🤖 Script 3: Drive-Struktur initialisieren (1:1 DEVKiTZ Spiegel)

Einmalig ausführen — erstellt exakt die gleiche Struktur wie C:\DEVKiTZ\

function initDriveStructure() {
  // Exakt gleiche Ordner wie C:\DEVKiTZ\
  const STRUCTURE = {
    '00_INBOX':     [],
    '01_PROJECTS':  [],
    '02_RESEARCH':  ['03_docs','ai-wiki','awesome-llm-skills',
                     'notebooklm','ai_chats','brain-archive'],
    '03_MEDIA':     ['Android-Fotos','Screenshots','Videos','Musik'],
    '04_SYSTEM':    ['00_Backups','ONTHERUN','SecondBrain',
                     'DEVKITZ_WIKI'],
    '05_INTERN':    [],
    '[DEEPKEEP]':   ['session-db','zip-backups','desktop-dokumente'],
    '[WORKSPACE]':  ['github-hub','templates'],
    '99_ARCHIVE':   ['auto-archived']
  };

  const root = DriveApp.getRootFolder();
  let created = 0;

  for (const [parent, children] of Object.entries(STRUCTURE)) {
    let pFolder = root.getFoldersByName(parent);
    pFolder = pFolder.hasNext()
      ? pFolder.next()
      : root.createFolder(parent);
    if (!pFolder) continue;

    children.forEach(child => {
      if (!pFolder.getFoldersByName(child).hasNext()) {
        pFolder.createFolder(child);
        created++;
        Logger.log('✅ Erstellt: ' + parent + '/' + child);
      }
    });
  }

  Logger.log('═══ Drive-Struktur: ' + created + ' Ordner erstellt ═══');
  Logger.log('Spiegel von C:\\DEVKiTZ\\ — fertig!');
}
