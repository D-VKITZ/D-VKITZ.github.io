// DkZ™ 4-Ordner-System — Google Drive Automatisierung
// Erstellt die 4 Ordner + Archiv und sortiert automatisch

function init4FolderSystem() {
  const FOLDERS = ['📋 TODO','📅 MEETING','💡 IDEA','🎯 USE_CASE','🗄️ ARCHIVE'];
  const root = DriveApp.getRootFolder();
  let base = root.getFoldersByName('DkZ_4-Ordner');
  base = base.hasNext() ? base.next() : root.createFolder('DkZ_4-Ordner');

  FOLDERS.forEach(name => {
    if (!base.getFoldersByName(name).hasNext()) {
      base.createFolder(name);
      Logger.log('✅ Erstellt: ' + name);
    }
  });
  Logger.log('═══ 4-Ordner-System bereit ═══');
}

// Zeit-Trigger: Alte Dateien → ARCHIVE (mit Verify!)
function autoArchiveTrigger() {
  const RULES = [
    { folder: '📋 TODO',     days: 7,  label: 'erledigt' },
    { folder: '📅 MEETING',  days: 3,  label: 'vergangen' },
    { folder: '💡 IDEA',     days: 30, label: 'unbearbeitet' },
    { folder: '🎯 USE_CASE', days: 90, label: 'inaktiv' }
  ];

  const base = findFolder('DkZ_4-Ordner');
  if (!base) return;
  const archive = base.getFoldersByName('🗄️ ARCHIVE').next();
  const now = new Date();
  let moved = 0;

  RULES.forEach(rule => {
    const src = base.getFoldersByName(rule.folder);
    if (!src.hasNext()) return;
    const folder = src.next();
    const files = folder.getFiles();

    while (files.hasNext()) {
      const f = files.next();
      const age = (now - f.getLastUpdated()) / 86400000;
      if (age > rule.days) {
        // COPY-VERIFY-REMOVE Pattern!
        archive.addFile(f);
        const verify = archive.getFilesByName(f.getName());
        if (verify.hasNext()) {
          folder.removeFile(f);
          moved++;
          log('🗄️ Archiviert (' + rule.label + ')', f.getName(), rule.folder);
        }
      }
    }
  });
  Logger.log('Archiviert: ' + moved + ' Dateien');
}

// Wöchentlicher Backup-Export
function weeklyBackup() {
  const base = findFolder('DkZ_4-Ordner');
  if (!base) return;
  const data = { exported: new Date().toISOString(), folders: {} };
  const subs = base.getFolders();

  while (subs.hasNext()) {
    const sub = subs.next();
    const files = sub.getFiles();
    data.folders[sub.getName()] = [];
    while (files.hasNext()) {
      const f = files.next();
      data.folders[sub.getName()].push({
        name: f.getName(), size: f.getSize(),
        modified: f.getLastUpdated().toISOString()
      });
    }
  }

  // Backup als JSON in DEEPKEEP
  const deepkeep = findOrCreate('[DEEPKEEP]/session-db');
  const filename = 'backup_4ordner_' + new Date().toISOString().slice(0,10) + '.json';
  deepkeep.createFile(filename, JSON.stringify(data, null, 2), 'application/json');
  Logger.log('💾 Backup: ' + filename);
}
